// Tracks how many times counter button is clicked
class Counter extends React.Component {
  // Constructs (adds state)
  constructor(props) {
    // Call parent constructor (Component)
    super(props);
    this.state = {
      clicks: 0
    };
  }

  // Renders
  render() {
    const {text} = this.props;
    const {clicks} = this.state;
    return (
      

      	{text}: {clicks}
  			
			

		)
	}

  // Increments counter on click
  // (Private method/property indicated with underscore)
  _onButtonClick = () => {
    this.setState({
      clicks: this.state.clicks + 1
    });
  };
}
